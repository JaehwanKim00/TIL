# SELECT,FROM,WHERE
SELECT 테이블의 어떤 컬럼을 선택할 것인가\
FROM 어떤 테이블에서 데이터를 확인할 것인가\
WHERE 원하는 조건\
전체 데이터 확인위해 SELECT *


# Pokemon Query
SELECT

 id as pokemon_id, # AS는 별칭을 지어줄 때 사용

 kor_name, # 컬럼이름에 따옴표 X

 eng_name,

 total

FROM `basic.pokemon`

where type1 = 'Fire'
